def calculate_swarm_damage(count):

** swarm-formation
#+BEGIN_SRC python
def swarm_formation(center_x, center_y, count, radius):
    import math
    return [(center_x + radius * math.cos(2*math.pi*i/count),
             center_y + radius * math.sin(2*math.pi*i/count))
            for i in range(count)]
#+END_SRC

** unit-strength
#+BEGIN_SRC python
def unit_strength(health, damage, armor):
    effective = damage * health / 100
    return effective * (1 - armor * 0.01)
#+END_SRC
